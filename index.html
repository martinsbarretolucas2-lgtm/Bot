const API_URL = "https://bot-ticket-cvbc.onrender.com/api/stats"; async function fetchStats() { try { const response = await fetch(API_URL); const data = await response.json(); // Isso preenche os números na tela document.getElementById('membros').innerText = data.membros; document.getElementById('tickets').innerText = data.ticketsAbertos; document.getElementById('ping').innerText = data.ping + "ms"; } catch (e) { console.error("Erro ao conectar com o Bot"); } } setInterval(fetchStats, 5000); // Atualiza a cada 5 segundos
